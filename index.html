import { serve } from "https://deno.land/std@0.168.0/http/server.ts" import { createClient } from "https://esm.sh/@supabase/supabase-js@2" const BASE_URL_FIELD = "https://amonamarth.fieldcontrol.com.br" const FIELD_API_KEY = Deno.env.get('FIELD_CONTROL_API_KEY') const SUPABASE_URL = Deno.env.get('SUPABASE_URL') ?? "" const SUPABASE_SERVICE_ROLE_KEY = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY') ?? "" const supabase = createClient(SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY) serve(async (req) => { const headers = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'GET, POST, OPTIONS', 'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type', 'Content-Type': 'application/json' } if (req.method === 'OPTIONS') return new Response('ok', { headers }) try { const data = await req.json() // 1. TRATAMENTO DE DADOS (Limpeza e Formatação) const cnpjLimpo = data.cnpj ? String(data.cnpj).replace(/\D/g, '').substring(0, 14) : "" const nomeCliente = String(data.cliente || "").toUpperCase().trim() const ticketStr = String(data.ticket || "000000") const [dia, mes, ano] = (data.data || "01/01/2026").split('/') const dataFormatadaField = `${ano}-${mes}-${dia}` const horaLimpa = data.slots && data.slots.length > 0 ? data.slots[0].substring(0, 5).padStart(5, '0') : "08:00" // Formatação do endereço para a mensagem da OS const enderecoFormatado = `${data.rua || "N/A"}, ${data.numero || "SN"} - ${data.bairro || "N/A"}, ${data.cidade || "N/A"}/${data.uf || "SP"}`; let locationId = ""; // --- PASSO A: BUSCA POR FILTRO EXATO DE CNPJ --- if (cnpjLimpo) { const resFiltro = await fetch(`${BASE_URL_FIELD}/locations?externalId=${cnpjLimpo}`, { headers: { 'X-Api-Key': FIELD_API_KEY || '' } }); const jsonFiltro = await resFiltro.json(); if (jsonFiltro.items && jsonFiltro.items.length > 0) { locationId = jsonFiltro.items[0].id; } } // --- PASSO B: BUSCA POR NOME (SE NÃO ACHOU POR CNPJ) --- if (!locationId) { const resNome = await fetch(`${BASE_URL_FIELD}/locations?search=${encodeURIComponent(nomeCliente)}`, { headers: { 'X-Api-Key': FIELD_API_KEY || '' } }); const jsonNome = await resNome.json(); const achado = jsonNome.items?.find(i => i.name.toUpperCase().trim() === nomeCliente); if (achado) locationId = achado.id; } // --- PASSO C: CADASTRO OU RESGATE POR CONFLITO --- if (!locationId) { const resCad = await fetch(`${BASE_URL_FIELD}/locations`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'X-Api-Key': FIELD_API_KEY || '' }, body: JSON.stringify({ "name": nomeCliente, "externalId": cnpjLimpo || null, "address": { "postalCode": (data.cep || "").replace(/\D/g, ''), "streetName": data.rua || "N/A", "number": String(data.numero || "SN"), "neighborhood": data.bairro || "N/A", "city": data.cidade || "N/A", "state": data.uf || "SP" }, "locationGroup": { "id": "fefada52-e087-4c0c-90a6-705e10d40f2a" } }) }); const dadosCad = await resCad.json(); if (resCad.ok) { locationId = dadosCad.id; } else if (resCad.status === 409 || JSON.stringify(dadosCad).includes('already')) { const resFinal = await fetch(`${BASE_URL_FIELD}/locations?limit=50&order=desc`, { headers: { 'X-Api-Key': FIELD_API_KEY || '' } }); const jsonFinal = await resFinal.json(); const resgatado = jsonFinal.items?.find(i => (cnpjLimpo && i.externalId && String(i.externalId).replace(/\D/g, '') === cnpjLimpo) || (i.name.toUpperCase().trim() === nomeCliente) ); locationId = resgatado?.id || jsonFinal.items?.[0]?.id; } } if (!locationId) throw new Error("Não foi possível obter o ID do cliente na Field Control."); // --- 2. SALVAR NO SUPABASE (TODOS OS CAMPOS) --- const { error: dbError } = await supabase.from('agendamentos').insert([{ ticket: ticketStr, cliente: nomeCliente, cnpj: cnpjLimpo || null, data: data.data, rawdate: data.rawdate || data.data || "", prestador: data.prestador || "", field_id: data.field_id || "", qtdtecnicos: parseInt(data.qtdtecnicos) || 1, solicitante: data.solicitante || "", slots: Array.isArray(data.slots) ? data.slots.join(', ') : (data.slots || ""), // Endereço completo rua: data.rua || "", numero: String(data.numero || ""), bairro: data.bairro || "", cidade: data.cidade || "", uf: data.uf || "", cep: data.cep || "", // Contato e extras acompanhamento: data.acompanhamento || "", telefone_responsavel: data.telefone_responsavel || "", segmento: data.segmento || "", // Tipo de Serviço: Extrai o nome exato (ex: Coblicam-Multi) tipo_servico: data.tipo_servico || (data.produtos_texto ? data.produtos_texto.split('] ')[1]?.split(' (Qtd:')[0] : ""), col5: data.col5 || "", // reserva produtos: data.produtos_texto || "" }]); if (dbError) throw new Error(`Erro Supabase: ${dbError.message}`); // --- 3. CRIAR ORDEM DE SERVIÇO NA FIELD --- const mensagemFinalField = ` 📦 SERVIÇOS SOLICITADOS: ${data.produtos_texto} 📍 ENDEREÇO DO ATENDIMENTO: ${enderecoFormatado} CEP: ${data.cep || "00000-000"} 👤 CONTATO NO LOCAL: ${data.acompanhamento || 'Não informado'} 📞 TELEFONE: ${data.telefone_responsavel || 'Não informado'} `.trim(); const resOS = await fetch(`${BASE_URL_FIELD}/maintenances`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'X-Api-Key': FIELD_API_KEY || '' }, body: JSON.stringify({ "number": ticketStr, "message": mensagemFinalField, "location": { "id": locationId }, "segment": { "id": data.segment_id_field }, "maintenanceType": { "id": data.maintenance_id_field }, "scheduleSuggestions": [{ "date": dataFormatadaField, "time": horaLimpa }] }) }).then(r => r.json()); if (!resOS.id) throw new Error(`Erro OS na Field: ${JSON.stringify(resOS)}`); return new Response(JSON.stringify({ success: true, fieldId: resOS.id }), { headers, status: 200 }); } catch (error) { return new Response(JSON.stringify({ error: error.message }), { headers, status: 400 }); } })
